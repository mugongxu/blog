``````js
var a = '12546498798783454466';

// 方法一：繁琐
a.toString().split('').reverse().join('').replace(/(\d{3})/g, '$1,').split('').reverse().join('').replace(/^,/, '');

// 方法二：简介
a.toString().replace(/(\d)(?=(?:\d{3})+$)/g, '$1,');

``````
